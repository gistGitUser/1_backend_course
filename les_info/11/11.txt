
Не особо много новой информации, но в gin можно
сразу валидировать поля

например можно указать что поле required
и должно иметь определённые значения (oneof=USD EUR)

type createAccountRequest struct {
	Owner    string `json:"owner" binding:"required"`
	Currency string `json:"currency" binding:"required,oneof=USD EUR"`
}

Помимо oneof можно установить min,max и интервалы min-max
для значений

type listAccountRequest struct {
	PageID   int32 `form:"page_id" binding:"required,min=1"`
	PageSize int32 `form:"page_size" binding:"required,min=5,max=10"`

	type getAccountRequest struct {
    	ID int64 `uri:"id" binding:"required,min=1"`
    }
